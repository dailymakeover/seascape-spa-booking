    <script>
        // Form submission handling
        document.getElementById('bookingForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Get form data
            const formData = new FormData(this);
            const data = {};
            
            // Convert FormData to regular object
            for (let [key, value] of formData.entries()) {
                if (data[key]) {
                    // Handle multiple values (like checkboxes)
                    if (Array.isArray(data[key])) {
                        data[key].push(value);
                    } else {
                        data[key] = [data[key], value];
                    }
                } else {
                    data[key] = value;
                }
            }
            
            // Create email content
            const subject = `Spa Booking Request - ${data.service || 'Service'} - ${data.name}`;
            const body = `
NEW SEASCAPE RESORT GUEST SPA BOOKING:

GUEST INFORMATION:
Name: ${data.name}
Room Number: ${data.room || 'Not provided'}
Room Number: ${data.room || 'Not provided'}
Email: ${data.email}
Phone: ${data.phone}
Check-out Date: ${data.checkout || 'Not provided'}
Referred By: ${data['referred-by'] || 'Not provided'}
Check-out Date: ${data.checkout || 'Not provided'}

SERVICE DETAILS:
Service: ${data.service}
Preferred Date: ${data.date}
Preferred Time: ${data.time}
Therapist Preference: ${data.therapist}

RESORT GUEST PACKAGES:
Group/Couples Interest: ${data['group-package'] ? 'YES - Guest interested in group/couples packages' : 'No'}
Returning Guest: ${data['repeat-guest'] ? 'YES - VIP returning guest' : 'No'}
Returning Guest: ${data['repeat-guest'] ? 'YES - VIP returning guest' : 'No'}

STAFF NOTES & REQUESTS:
${data.notes || 'None'}

PRIVACY COMPLIANCE:
Guest Consent: ${data['privacy-consent'] ? 'YES - Guest has consented to data collection and use' : 'NO CONSENT GIVEN'}
Data Protection: Guest information handled per The Daily Makeover privacy policy

BOOKING DETAILS:
Booked by Staff: ${data['staff-name'] || 'Not provided'}
Booking Time: ${data['booking-time'] || 'Not provided'}

ACTION REQUIRED: Contact guest within 2 hours to confirm in-room spa appointment.

---
Sent from The Daily Makeover Spa Booking System - Seascape Resort Partnership
Privacy Compliant Booking - Guest consent verified
            `;
            
            // Create mailto link
            const mailtoLink = `mailto:support@thedailymakeover.com?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;
            
            // Open email client
            window.location.href = mailtoLink;
            
            // Show success message
            document.getElementById('successMessage').style.display = 'block';
            document.getElementById('successMessage').innerHTML = 'Guest booking submitted successfully! Spa team will contact the guest within 2 hours.';
            document.getElementById('bookingForm').style.display = 'none';
            
            // Scroll to top
            window.scrollTo(0, 0);
        });

        // Radio button styling
        document.querySelectorAll('input[type="radio"]').forEach(radio => {
            radio.addEventListener('change', function() {
                // Remove selected class from all options in this group
                document.querySelectorAll(`input[name="${this.name}"]`).forEach(r => {
                    r.closest('.radio-option').classList.remove('selected');
                });
                // Add selected class to chosen option
                this.closest('.radio-option').classList.add('selected');
            });
        });

        // Checkbox styling
        document.querySelectorAll('input[type="checkbox"]').forEach(checkbox => {
            checkbox.addEventListener('change', function() {
                if (this.checked) {
                    this.closest('.checkbox-option').classList.add('checked');
                } else {
                    this.closest('.checkbox-option').classList.remove('checked');
                }
            });
        });

        // Set minimum date to today
        document.getElementById('date').min = new Date().toISOString().split('T')[0];

        // Initialize radio button states
        document.addEventListener('DOMContentLoaded', function() {
            document.querySelectorAll('input[type="radio"]:checked').forEach(radio => {
                radio.closest('.radio-option').classList.add('selected');
            });
            
            // Initialize submit button state
            updateSubmitButton();
        });

        // Consent checkbox functionality
        function toggleConsent() {
            const checkbox = document.getElementById('privacy-consent');
            const container = checkbox.closest('.consent-checkbox');
            
            checkbox.checked = !checkbox.checked;
            
            if (checkbox.checked) {
                container.classList.add('checked');
            } else {
                container.classList.remove('checked');
            }
            
            updateSubmitButton();
        }

        // Update submit button based on consent
        function updateSubmitButton() {
            const consentCheckbox = document.getElementById('privacy-consent');
            const submitButton = document.querySelector('.submit-btn');
            
            if (consentCheckbox.checked) {
                submitButton.disabled = false;
                submitButton.textContent = 'Send Booking Request';
            } else {
                submitButton.disabled = true;
                submitButton.textContent = 'Please Accept Privacy Terms';
            }
        }

        // Listen for consent changes
        document.getElementById('privacy-consent').addEventListener('change', updateSubmitButton);
